
include_directories (${CMAKE_SOURCE_DIR}/3rd_party/asio/include)

  Server.cpp
  Session.cpp
	commands.cpp
)

set(net_HEADERS
  Server.h
  Session.h
	Command.h
	commands.h
)

# bbproto
include_directories(${CMAKE_SOURCE_DIR}/bbproto/asn.1)

file(GLOB files "${CMAKE_SOURCE_DIR}/bbproto/asn.1/*.c") # these files are generated by asn1c from trace_proto.asn1 

set_source_files_properties(${files} PROPERTIES LANGUAGE CXX) 

add_definitions("-DASIO_STANDALONE")
add_definitions("-DASSUMESTDTYPES") # asn1: use types from cstdint 
add_definitions("-DUSE_CXX_ALLOCATOR") # asn1: use custom c++ allocator for encoder / decoder 
add_definitions("-D_CRT_SECURE_NO_WARNINGS") # msvc shut up pls

add_library(net STATIC ${net_SOURCES} ${net_HEADERS} ${files})
